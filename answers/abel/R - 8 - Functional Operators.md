2. Extend and() and or() to deal with any number of input functions. Can you do it with Reduce()? Can you keep them lazy (e.g., for and(), the function returns once it sees the first FALSE)?

f_and <- function(e1, e2) {
  force(e1)
  force(e2)
  function(...) {e1(...) && e2(...)}
}

f_or <- function(e1, e2) {
  force(e1)
  force(e2)
  function(...) {e1(...) || e2(...)}
}
